toggle Toggle ⚠ Do NOT call UpdateToggle in a loop
text string info string callback(state) function 
Section:NewToggle("Speed Hack", "Toggle walk speed", function(state)
    local hum = game.Players.LocalPlayer.Character
        and game.Players.LocalPlayer.Character:FindFirstChild("Humanoid")
    if hum then hum.WalkSpeed = state and 64 or 16 end
end)

color pickers Full Live Theme Editor (loop) 
local myTheme = {
    SchemeColor  = Color3.fromRGB(26, 189, 158),
    Background   = Color3.fromRGB(44, 62,  82),
    Header       = Color3.fromRGB(57, 81,  105),
    TextColor    = Color3.fromRGB(255, 255, 255),
    ElementColor = Color3.fromRGB(52, 74,  95),
}
for key, color in pairs(myTheme) do
    Section:NewColorPicker(key, "Change "..key, color, function(c)
        Library:ChangeColor(key, c)
    end)
end
Full Script Template
template Copy-paste starter script 
-- Load library
local Library = loadstring(game:HttpGet(
    "https://raw.githubusercontent.com/yotolishol-sketch/Ui-library-Kavo-/502f4b61c3a14f1d0e49738c0ff532a9e077662d/source.lua.txt"
))()

-- Create window
local Window = Library.CreateLib("My Script", "Midnight")

-- Tab + Section
local Tab     = Window:NewTab("Main")
local Section = Tab:NewSection("General")

-- Label
Section:NewLabel("Welcome!")

-- Button
Section:NewButton("Do Something", "Press to run", function()
    print("Button pressed")
end)

-- Toggle
Section:NewToggle("Speed Hack", "Toggle speed", function(state)
    local hum = game.Players.LocalPlayer.Character
        and game.Players.LocalPlayer.Character:FindFirstChild("Humanoid")
    if hum then hum.WalkSpeed = state and 64 or 16 end
end)

-- Slider
Section:NewSlider("Walk Speed", "Adjust speed", 200, 16, function(v)
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = v
end)

-- Keybind to hide/show UI
Section:NewKeybind("Toggle UI", "RightShift", Enum.KeyCode.RightShift, function()
    Library:ToggleUI()
end)
